def locate_line(src: SourceDoc, pattern, nth=1):
    # returns line number and line content of nth match
    matches=[]
    for i,l in enumerate(src.lines, start=1):
        if re.search(pattern, l):
            matches.append((i,l))
    if len(matches)>=nth:
        return matches[nth-1]
    return None

def find_docindex_entry(path_substring):
    # find the heading line '### ...<path_substring>' and then the next purpose line
    lines=sources["DOC_INDEX"].lines
    idx=None
    for i,l in enumerate(lines, start=1):
        if l.startswith("###") and path_substring in l:
            idx=i
            break
    if idx is None:
        return None
    # find nearest '- purpose' within next 15 lines
    purpose=None
    for j in range(idx, min(idx+20, len(lines)+1)):
        if "purpose (extracted):" in lines[j-1]:
            purpose=(j, lines[j-1])
            break
    return {"heading": (idx, lines[idx-1]), "purpose": purpose}
